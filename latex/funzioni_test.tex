\section*{Introduzione}

Questo documento raccoglie tutte le funzioni di test utilizzate nell'articolo di Grippo, Lampariello e Lucidi (1991) sull'ottimizzazione non vincolata con metodi di stabilizzazione non monotoni. Per ogni funzione sono stati calcolati analiticamente il gradiente e la matrice hessiana, strumenti fondamentali per l'implementazione di metodi di ottimizzazione di tipo Newton.

Le funzioni sono organizzate in due sezioni:
\begin{itemize}
    \item Problemi 1-18 dalla raccolta standard di More et al. (1981)
    \item Problemi 19-33 dall'appendice del paper
\end{itemize}

Per ogni funzione vengono forniti:
\begin{itemize}
    \item L'espressione analitica della funzione
    \item Il punto iniziale $x_0$ per i test numerici
    \item Il punto di minimo noto $x^*$ (quando disponibile)
    \item Il valore della funzione nel minimo $f(x^*)$
    \item Il gradiente $\nabla f(x)$
    \item La matrice hessiana $H(x)$
\end{itemize}

\section*{Problemi 1-18 da More et al. (1981)}

\subsection*{Problema 1: Rosenbrock Function}
\[
f(x) = 100(x_2 - x_1^2)^2 + (1 - x_1)^2
\]
\[
x_0 = [-1.2, 1]^T, \quad x^* = [1, 1]^T, \quad f(x^*) = 0.
\]

\textbf{Gradiente:}
\[
\nabla f(x) = \begin{bmatrix}
-400x_1(x_2 - x_1^2) - 2(1 - x_1) \\
200(x_2 - x_1^2)
\end{bmatrix}
\]

\textbf{Hessiano:}
\[
H(x) = \begin{bmatrix}
1200x_1^2 - 400x_2 + 2 & -400x_1 \\
-400x_1 & 200
\end{bmatrix}
\]

\subsection*{Problema 2: Powell Badly Scaled Function}
\[
f(x) = (10^4 x_1 x_2 - 1)^2 + (e^{-x_1} + e^{-x_2} - 1.0001)^2
\]
\[
x_0 = [0, 1]^T, \quad x^* \approx [1.098\times 10^{-5}, 9.106]^T, \quad f(x^*) = 0.
\]

\textbf{Gradiente:}
\[
\nabla f(x) = \begin{bmatrix}
2(10^4 x_1 x_2 - 1)10^4 x_2 - 2(e^{-x_1} + e^{-x_2} - 1.0001)e^{-x_1} \\
2(10^4 x_1 x_2 - 1)10^4 x_1 - 2(e^{-x_1} + e^{-x_2} - 1.0001)e^{-x_2}
\end{bmatrix}
\]

\textbf{Hessiano:}
\[
H(x) = \begin{bmatrix}
2(10^4 x_2)^2 + 2e^{-2x_1} + 2A e^{-x_1} & 2\cdot10^8 x_1 x_2 + 2\cdot10^4 - 2e^{-x_1}e^{-x_2} \\
2\cdot10^8 x_1 x_2 + 2\cdot10^4 - 2e^{-x_1}e^{-x_2} & 2(10^4 x_1)^2 + 2e^{-2x_2} + 2A e^{-x_2}
\end{bmatrix}
\]
dove $A = e^{-x_1} + e^{-x_2} - 1.0001$.

\subsection*{Problema 3: Brown Badly Scaled Function}
\[
f(x) = (x_1 - 10^6)^2 + (x_2 - 2\times10^{-6})^2 + (x_1 x_2 - 2)^2
\]
\[
x_0 = [1, 1]^T, \quad x^* = [10^6, 2\times10^{-6}]^T, \quad f(x^*) = 0.
\]

\textbf{Gradiente:}
\[
\nabla f(x) = \begin{bmatrix}
2(x_1 - 10^6) + 2(x_1 x_2 - 2)x_2 \\
2(x_2 - 2\times10^{-6}) + 2(x_1 x_2 - 2)x_1
\end{bmatrix}
\]

\textbf{Hessiano:}
\[
H(x) = \begin{bmatrix}
2 + 2x_2^2 & 4x_1 x_2 - 4 \\
4x_1 x_2 - 4 & 2 + 2x_1^2
\end{bmatrix}
\]

\subsection*{Problema 4: Beale Function}
\[
f(x) = [1.5 - x_1(1 - x_2)]^2 + [2.25 - x_1(1 - x_2^2)]^2 + [2.625 - x_1(1 - x_2^3)]^2
\]
\[
x_0 = [1, 1]^T, \quad x^* = [3, 0.5]^T, \quad f(x^*) = 0.
\]

\textbf{Gradiente:}
\begin{align*}
\frac{\partial f}{\partial x_1} &= -2(1 - x_2)[1.5 - x_1(1 - x_2)] - 2(1 - x_2^2)[2.25 - x_1(1 - x_2^2)] \\
&\quad - 2(1 - x_2^3)[2.625 - x_1(1 - x_2^3)] \\
\frac{\partial f}{\partial x_2} &= 2x_1[1.5 - x_1(1 - x_2)] + 4x_1 x_2[2.25 - x_1(1 - x_2^2)] \\
&\quad + 6x_1 x_2^2[2.625 - x_1(1 - x_2^3)]
\end{align*}

\textbf{Hessiano:}
\begin{align*}
\frac{\partial^2 f}{\partial x_1^2} &= 2(1 - x_2)^2 + 2(1 - x_2^2)^2 + 2(1 - x_2^3)^2 \\
\frac{\partial^2 f}{\partial x_1 \partial x_2} &= -2[1.5 - 2x_1(1 - x_2)] + 4x_1 x_2[2.25 - 2x_1(1 - x_2^2)] \\
&\quad + 6x_1 x_2^2[2.625 - 2x_1(1 - x_2^3)] \\
\frac{\partial^2 f}{\partial x_2^2} &= 2x_1^2 + 8x_1^2 x_2^2 + 18x_1^2 x_2^4
\end{align*}

\subsection*{Problema 5: Gulf Research and Development Function}
\[
f(x) = \sum_{i=1}^{m} \left[ e^{-(u_i - x_2)^{x_3}/x_1} - t_i \right]^2, \quad u_i = 25 + [-50 \ln(t_i)]^{2/3}
\]
\[
t_i = 0.1i, \quad m = 3, \quad x_0 = [5, 2.5, 0.15]^T, \quad x^* = [50, 25, 1.5]^T, \quad f(x^*) = 0.
\]

\textbf{Gradiente:} (forma compatta)
\begin{align*}
\frac{\partial f}{\partial x_1} &= \sum_{i=1}^{m} 2\left[ e^{-(u_i - x_2)^{x_3}/x_1} - t_i \right] \cdot \frac{(u_i - x_2)^{x_3}}{x_1^2} e^{-(u_i - x_2)^{x_3}/x_1} \\
\frac{\partial f}{\partial x_2} &= \sum_{i=1}^{m} 2\left[ e^{-(u_i - x_2)^{x_3}/x_1} - t_i \right] \cdot \frac{x_3 (u_i - x_2)^{x_3-1}}{x_1} e^{-(u_i - x_2)^{x_3}/x_1} \\
\frac{\partial f}{\partial x_3} &= \sum_{i=1}^{m} 2\left[ e^{-(u_i - x_2)^{x_3}/x_1} - t_i \right] \cdot \frac{(u_i - x_2)^{x_3} \ln(u_i - x_2)}{x_1} e^{-(u_i - x_2)^{x_3}/x_1}
\end{align*}

\subsection*{Problema 6: Helical Valley Function}
\[
f(x) = 100[x_3 - 10\theta(x_1,x_2)]^2 + [\sqrt{x_1^2 + x_2^2} - 1]^2 + x_3^2
\]
\[
\theta(x_1,x_2) = \begin{cases}
\frac{1}{2\pi} \arctan(x_2/x_1) & \text{se } x_1 > 0 \\
\frac{1}{2\pi} \arctan(x_2/x_1) + 0.5 & \text{se } x_1 < 0
\end{cases}
\]
\[
x_0 = [-1, 0, 0]^T, \quad x^* = [1, 0, 0]^T, \quad f(x^*) = 0.
\]

\textbf{Gradiente:}
\begin{align*}
\frac{\partial f}{\partial x_1} &= -2000[x_3 - 10\theta]\frac{\partial \theta}{\partial x_1} + 2(\sqrt{x_1^2 + x_2^2} - 1)\frac{x_1}{\sqrt{x_1^2 + x_2^2}} \\
\frac{\partial f}{\partial x_2} &= -2000[x_3 - 10\theta]\frac{\partial \theta}{\partial x_2} + 2(\sqrt{x_1^2 + x_2^2} - 1)\frac{x_2}{\sqrt{x_1^2 + x_2^2}} \\
\frac{\partial f}{\partial x_3} &= 200[x_3 - 10\theta] + 2x_3
\end{align*}
dove $\frac{\partial \theta}{\partial x_1} = -\frac{x_2}{2\pi(x_1^2 + x_2^2)}$, $\frac{\partial \theta}{\partial x_2} = \frac{x_1}{2\pi(x_1^2 + x_2^2)}$.

\subsection*{Problema 7: Bard Function}
\[
f(x) = \sum_{i=1}^{15} \left[ y_i - x_1 - \frac{u_i}{x_2 v_i + x_3 w_i} \right]^2
\]
dove $u_i = i$, $v_i = 16 - i$, $w_i = \min(u_i, v_i)$, $y_i$ sono dati.
\[
x_0 = [1, 1, 1]^T, \quad x^* \approx [0.082, 1.133, 2.344]^T.
\]

\textbf{Gradiente:}
\begin{align*}
\frac{\partial f}{\partial x_1} &= -2\sum_{i=1}^{15} \left[ y_i - x_1 - \frac{u_i}{x_2 v_i + x_3 w_i} \right] \\
\frac{\partial f}{\partial x_2} &= 2\sum_{i=1}^{15} \left[ y_i - x_1 - \frac{u_i}{x_2 v_i + x_3 w_i} \right] \frac{u_i v_i}{(x_2 v_i + x_3 w_i)^2} \\
\frac{\partial f}{\partial x_3} &= 2\sum_{i=1}^{15} \left[ y_i - x_1 - \frac{u_i}{x_2 v_i + x_3 w_i} \right] \frac{u_i w_i}{(x_2 v_i + x_3 w_i)^2}
\end{align*}

\subsection*{Problema 8: Gaussian Function}
\[
f(x) = \sum_{i=1}^{15} \left[ x_1 e^{-x_2 (t_i - x_3)^2/2} - y_i \right]^2, \quad t_i = (8 - i)/2
\]
\[
y_i \text{ sono dati}, \quad x_0 = [0.4, 1, 0]^T, \quad x^* \approx [0.4, 1, 0]^T.
\]

\textbf{Gradiente:}
\begin{align*}
\frac{\partial f}{\partial x_1} &= 2\sum_{i=1}^{15} \left[ x_1 e^{-x_2 (t_i - x_3)^2/2} - y_i \right] e^{-x_2 (t_i - x_3)^2/2} \\
\frac{\partial f}{\partial x_2} &= -2\sum_{i=1}^{15} \left[ x_1 e^{-x_2 (t_i - x_3)^2/2} - y_i \right] x_1 \frac{(t_i - x_3)^2}{2} e^{-x_2 (t_i - x_3)^2/2} \\
\frac{\partial f}{\partial x_3} &= 2\sum_{i=1}^{15} \left[ x_1 e^{-x_2 (t_i - x_3)^2/2} - y_i \right] x_1 x_2 (t_i - x_3) e^{-x_2 (t_i - x_3)^2/2}
\end{align*}

\subsection*{Problema 9: Box 3-Dimensional Function}
\[
f(x) = \sum_{i=1}^{10} \left[ e^{-t_i x_1} - e^{-t_i x_2} - x_3(e^{-t_i} - e^{-10t_i}) \right]^2, \quad t_i = 0.1i
\]
\[
x_0 = [0, 10, 20]^T, \quad x^* = [1, 10, 1]^T, \quad f(x^*) = 0.
\]

\textbf{Gradiente:}
\begin{align*}
\frac{\partial f}{\partial x_1} &= -2\sum_{i=1}^{10} \left[ e^{-t_i x_1} - e^{-t_i x_2} - x_3(e^{-t_i} - e^{-10t_i}) \right] t_i e^{-t_i x_1} \\
\frac{\partial f}{\partial x_2} &= 2\sum_{i=1}^{10} \left[ e^{-t_i x_1} - e^{-t_i x_2} - x_3(e^{-t_i} - e^{-10t_i}) \right] t_i e^{-t_i x_2} \\
\frac{\partial f}{\partial x_3} &= -2\sum_{i=1}^{10} \left[ e^{-t_i x_1} - e^{-t_i x_2} - x_3(e^{-t_i} - e^{-10t_i}) \right] (e^{-t_i} - e^{-10t_i})
\end{align*}

\subsection*{Problema 10: Powell Singular Function}

\subsection*{Problema 11: Wood Function}

\subsection*{Problema 12: Kowalik and Osborne Function}
\[
f(x) = \sum_{i=1}^{11} \left[ \frac{x_1(u_i^2 + u_i x_2)}{u_i^2 + u_i x_3 + x_4} - y_i \right]^2
\]
dove $u_i = 0.1(i-1)$, $y_i$ sono dati.
\[
x_0 = [0.25, 0.39, 0.415, 0.39]^T, \quad x^* \approx [0.192, 0.190, 0.123, 0.136]^T.
\]

\textbf{Gradiente:}
\begin{align*}
\frac{\partial f}{\partial x_1} &= 2\sum_{i=1}^{11} \left[ \frac{x_1(u_i^2 + u_i x_2)}{u_i^2 + u_i x_3 + x_4} - y_i \right] \frac{u_i^2 + u_i x_2}{u_i^2 + u_i x_3 + x_4} \\
\frac{\partial f}{\partial x_2} &= 2\sum_{i=1}^{11} \left[ \frac{x_1(u_i^2 + u_i x_2)}{u_i^2 + u_i x_3 + x_4} - y_i \right] \frac{x_1 u_i}{u_i^2 + u_i x_3 + x_4} \\
\frac{\partial f}{\partial x_3} &= -2\sum_{i=1}^{11} \left[ \frac{x_1(u_i^2 + u_i x_2)}{u_i^2 + u_i x_3 + x_4} - y_i \right] \frac{x_1 u_i (u_i^2 + u_i x_2)}{(u_i^2 + u_i x_3 + x_4)^2} \\
\frac{\partial f}{\partial x_4} &= -2\sum_{i=1}^{11} \left[ \frac{x_1(u_i^2 + u_i x_2)}{u_i^2 + u_i x_3 + x_4} - y_i \right] \frac{x_1 (u_i^2 + u_i x_2)}{(u_i^2 + u_i x_3 + x_4)^2}
\end{align*}

\subsection*{Problema 13: Brown and Dennis Function}
\[
f(x) = \sum_{i=1}^{20} \left[ (x_1 + t_i x_2 - e^{t_i})^2 + (x_3 + x_4 \sin(t_i) - \cos(t_i))^2 \right], \quad t_i = 0.2i
\]
\[
x_0 = [25, 5, -5, -1]^T, \quad x^* \approx [-11.59, 13.59, -0.4, 0.3]^T.
\]

\textbf{Gradiente:}
\begin{align*}
\frac{\partial f}{\partial x_1} &= 2\sum_{i=1}^{20} (x_1 + t_i x_2 - e^{t_i}) \\
\frac{\partial f}{\partial x_2} &= 2\sum_{i=1}^{20} t_i(x_1 + t_i x_2 - e^{t_i}) \\
\frac{\partial f}{\partial x_3} &= 2\sum_{i=1}^{20} (x_3 + x_4 \sin(t_i) - \cos(t_i)) \\
\frac{\partial f}{\partial x_4} &= 2\sum_{i=1}^{20} \sin(t_i)(x_3 + x_4 \sin(t_i) - \cos(t_i))
\end{align*}

\textbf{Hessiano:}
\[
H(x) = \begin{bmatrix}
40 & 2\sum t_i & 0 & 0 \\
2\sum t_i & 2\sum t_i^2 & 0 & 0 \\
0 & 0 & 40 & 2\sum \sin(t_i) \\
0 & 0 & 2\sum \sin(t_i) & 2\sum \sin^2(t_i)
\end{bmatrix}
\]

\subsection*{Problema 14: Biggs EXP6 Function}

\subsection*{Problema 16: Extended Rosenbrock Function}

\section*{Problemi 19-33 dall'Appendice}

\subsection*{Problema 19: Scaled Rosenbrock Function}
\[
f(x) = c(x_2 - x_1^2)^2 + (1 - x_1)^2
\]
\[
x_0 = [-1.2, 1]^T, \quad x^* = [1, 1]^T, \quad f(x^*) = 0.
\]

\textbf{Gradiente:}
\[
\nabla f(x) = \begin{bmatrix}
-4c(x_2 - x_1^2)x_1 - 2(1 - x_1) \\
2c(x_2 - x_1^2)
\end{bmatrix}
\]
